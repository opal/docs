def cp source, target
  target = File.join(target, File.basename(source)) if File.directory? target
  %x`__fs__.writeFileSync(target, __fs__.readFileSync(source));`
end

def mkdir_p path

  %x`__fs__.mkdirSync(#{path})`
end
