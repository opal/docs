def compile
  name, base = name_and_base
  helper :klass

  push "(function($base, $super) {"
  line "  function $#{name}(){};"
  line "  var self = $#{name} = $klass($base, $super, '#{name}', $#{name});"

  in_scope do
    scope.name = name
    add_temp "#{scope.proto} = self._proto"
    add_temp "$scope = self._scope"

    body_code = self.body_code
    empty_line

    line scope.to_vars
    line body_code
  end

  line "})(", base, ", ", self.super_code, ")"
end
