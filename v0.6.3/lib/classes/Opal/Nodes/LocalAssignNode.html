def compile
  if using_irb?
    push "$opal.irb_vars#{property var_name.to_s} = "
    push expr(value)
  else
    add_local variable(var_name.to_s)

    push "#{variable(var_name.to_s)} = "
    push expr(value)
  end

  wrap '(', ')' if recv?
end
