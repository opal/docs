def initialize(buffer, offset = nil, length = nil)
  if native?(buffer)
    super(buffer)
  elsif offset && length
    super(%x`new DataView(#{buffer.to_n}, #{offset.to_n}, #{length.to_n})`)
  elsif offset
    super(%x`new DataView(#{buffer.to_n}, #{offset.to_n})`)
  else
    super(%x`new DataView(#{buffer.to_n})`)
  end

  @buffer = buffer
  @offset = offset
end
