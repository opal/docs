def initialize(regexp, match_groups)
  $~          = self
  @regexp     = regexp
  @begin      = %x`match_groups.index`
  @string     = %x`match_groups.input`
  @pre_match  = %x`#@string.substr(0, regexp.lastIndex - match_groups[0].length)`
  @post_match = %x`#@string.substr(regexp.lastIndex)`
  @matches    = []

  %x{
    for (var i = 0, length = match_groups.length; i < length; i++) {
      var group = match_groups[i];

      if (group == null) {
        #@matches.push(nil);
      }
      else {
        #@matches.push(group);
      }
    }
  }
end

def ==(other)

  %x`self.regexp == other.regexp` &&
  %x`self.pre_match == other.pre_match` &&
  %x`self.post_match == other.post_match` &&
  %x`self.begin == other.begin`
end

def inspect
  %x{
    var str = "#<MatchData " + #{`#@matches[0]`.inspect};

    for (var i = 1, length = #@matches.length; i < length; i++) {
      str += " " + i + ":" + #{`#@matches[i]`.inspect};
    }

    return str + ">";
  }
end

def values_at(*indexes)
  %x{
    var values       = [],
        match_length = #@matches.length;

    for (var i = 0, length = indexes.length; i < length; i++) {
      var pos = indexes[i];

      if (pos >= 0) {
        values.push(#@matches[pos]);
      }
      else {
        pos += match_length;

        if (pos > 0) {
          values.push(#@matches[pos]);
        }
        else {
          values.push(nil);
        }
      }
    }

    return values;
  }
