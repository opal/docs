# File opal/stdlib/minitest/pride_plugin.rb, line 89
def pride string
  string = "*" if string == "."
  c = @colors[@index % @size]
  @index += 1
  "#{ESC}#{c}m#{string}#{NND}"
end

# File opal/stdlib/minitest/pride_plugin.rb, line 60
def print o
  case o
  when "." then
    io.print pride o
  when "E", "F" then
    io.print "#{ESC}41m#{ESC}37m#{o}#{NND}"
  when "S" then
    io.print pride o
  else
    io.print o
  end
end
