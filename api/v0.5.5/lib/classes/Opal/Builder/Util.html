# File opal/lib/opal/builder.rb, line 114
def gzip(str)
  IO.popen('gzip -f 2> /dev/null', 'r+') do |i|
    i.puts str
    i.close_write
    return i.read
  end
rescue Errno::ENOENT
  $stderr.puts '"gzip" command not found, it is required to produce the .gz version'
  nil
end
