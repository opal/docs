def compile(source, file_name = '(erb)')
  @source, @file_name, @result = source, file_name, source

  self.fix_quotes
  self.find_contents
  self.find_code
  self.wrap_compiled

  Opal.compile @result
end

def find_contents
  @result = @result.gsub(/<%=([\s\S]+?)%>/) do
    inner = $1.gsub(/\'/, "'").gsub(/\"/, '"')

    if inner =~ BLOCK_EXPR
      "\")\noutput_buffer.append= #{ inner }\noutput_buffer.append(\""
    else
      "\")\noutput_buffer.append=(#{ inner })\noutput_buffer.append(\""
    end
  end
end
