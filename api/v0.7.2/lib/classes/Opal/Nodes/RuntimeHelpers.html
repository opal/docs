def self.helper(name, &block)
  HELPERS << name
  define_method("compile_#{name}", &block)
end

def compile
  if HELPERS.include?(meth.to_sym)
    __send__("compile_#{meth}")
  else
    raise "Helper not supported: #{meth}"
  end
end
