def compile_split_lines(value, sexp)
  idx = 0
  value.each_line do |line|
    if idx == 0
      push line
    else
      line_sexp = s()
      line_sexp.source = [sexp.line + idx, 0]
      frag = Fragment.new(line, line_sexp)
      push frag
    end

    idx += 1
  end
end
