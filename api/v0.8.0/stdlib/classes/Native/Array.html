def initialize(native, options = {}, &block)
  super(native)

  @get    = options[:get] || options[:access]
  @named  = options[:named]
  @set    = options[:set] || options[:access]
  @length = options[:length] || :length
  @block  = block

  if %x`#{length} == null`
    raise ArgumentError, "no length found on the array-like object"
  end
end

def [](index)
  result = case index
    when String, Symbol
      @named ? %x`#@native[#@named](#{index})` : %x`#@native[#{index}]`

    when Integer
      @get ? %x`#@native[#@get](#{index})` : %x`#@native[#{index}]`
  end

  if result
    if @block
      @block.call(result)
    else
      Native(result)
    end
  end
end

def []=(index, value)
  if @set
    %x`#@native[#@set](#{index}, #{Native.convert(value)})`
  else
    %x`#@native[#{index}] = #{Native.convert(value)}`
  end
end

def each(&block)
  return enum_for :each unless block

  %x{
    for (var i = 0, length = #{length}; i < length; i++) {
      var value = Opal.yield1(block, #{self[`i`]});

      if (value === $breaker) {
        return $breaker.$v;
      }
    }
  }

  self
end

def last(count = nil)
  if count
    index  = length - 1
    result = []

    while index >= 0
      result << self[index]
      index  -= 1
    end

    result
  else
    self[length - 1]
  end
end
