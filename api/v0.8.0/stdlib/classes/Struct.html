def initialize(*args)
  if args.length == 1 && native?(args[0])
    object = args[0]

    members.each {|name|
      instance_variable_set "@#{name}", Native(%x`#{object}[#{name}]`)
    }
  else
    members.each_with_index {|name, index|
      instance_variable_set "@#{name}", args[index]
    }
  end
end

def to_n
  result = %x`{}`

  each_pair {|name, value|
    %x`#{result}[#{name}] = #{value.to_n}`
  }

  result
end
