            value = Opal.yield1(block, param);

          result = $breaker.$v;
          return $breaker;

        all.push(param);

        return result;
      }

      if (all.length === 0) {

            }

          if (Opal.yield1(block, slice) === $breaker) {
            result = $breaker.$v;
            return $breaker;
          }

          slice = [];
        }
      };

      self.$each();

      if (result !== undefined) {
        return result;

      // our "last" group, if smaller than n then won't have been yielded

          return $breaker.$v;
        }
      }
    }

    nil
  end
