def initialize(regexp, match_groups)
  $~          = self
  @regexp     = regexp
  @begin      = %x`match_groups.index`
  @string     = %x`match_groups.input`
  @pre_match  = %x`match_groups.input.slice(0, match_groups.index)`
  @post_match = %x`match_groups.input.slice(match_groups.index + match_groups[0].length)`
  @matches    = []

  %x{
    for (var i = 0, length = match_groups.length; i < length; i++) {
      var group = match_groups[i];

      if (group == null) {
        #@matches.push(nil);
      }
      else {
        #@matches.push(group);
      }
    }
  }
end

def ==(other)

  %x`self.regexp.toString() == other.regexp.toString()` &&
  %x`self.pre_match == other.pre_match` &&
  %x`self.post_match == other.post_match` &&
  %x`self.begin == other.begin`
end

def values_at(*args)
  %x{
    var i, a, index, values = [];

    for (i = 0; i < args.length; i++) {

      if (args[i].$$is_range) {
        a = #{`args[i]`.to_a};
        a.unshift(i, 1);
        Array.prototype.splice.apply(args, a);
      }

      index = #{Opal.coerce_to!(`args[i]`, Integer, :to_int)};

      if (index < 0) {
        index += #@matches.length;
        if (index < 0) {
          values.push(nil);
          continue;
        }
      }

      values.push(#@matches[index]);
    }

    return values;
  }
end
