def expand(path)
  if Pathname.new(path).absolute? || path =~ %r{\A\.?\.#{File::SEPARATOR}}
    path
  else
    file_finder.find(path)
  end
end

def read(path)
  full_path = expand(path)
  return nil if full_path.nil?
  File.open(full_path, 'rb:UTF-8'){|f| f.read}
end
