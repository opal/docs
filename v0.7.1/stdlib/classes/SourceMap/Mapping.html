def inspect
  str = "#<#{self.class} source=#{source.inspect}"
  str << " generated=#{generated}, original=#{original}"
  str << " name=#{name.inspect}" if name
  str << ">"
  str
end

def to_s
  str = "#{generated.line}:#{generated.column}"
  str << "->#{source}@#{original.line}:#{original.column}"
  str << "##{name}" if name
  str
end
