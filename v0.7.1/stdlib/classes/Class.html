def native_alias(new_jsid, existing_mid)
  %x{
    var aliased = #{self}.$$proto['$' + #{existing_mid}];
    if (!aliased) {
      #{raise NameError, "undefined method `#{existing_mid}' for class `#{inspect}'"};
    }
    #{self}.$$proto[#{new_jsid}] = aliased;
  }
end
