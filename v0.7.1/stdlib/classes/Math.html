def acos(x)
  %x{
    if (!#{Numeric === x}) {
      #{raise Opal.type_error(x, Float)};
    }

    x = #{x.to_f};

    if (x < -1 || x > 1) {
      #{raise DomainError, :acos};
    }

    return Math.acos(x);
  }
end

def log(num, base = E, method = nil)
  %x{
    if (!#{Numeric === num}) {
      #{raise Opal.type_error(num, Float)};
    }

    if (!#{Numeric === base}) {
      #{raise Opal.type_error(base, Float)};
    }

    num  = #{num.to_f};
    base = #{base.to_f};

    if (num < 0) {
      #{raise DomainError, method || :log};
    }

    num = Math.log(num);

    if (base != Math.E) {
      num /= Math.log(base);
    }

    return num
  }
end
