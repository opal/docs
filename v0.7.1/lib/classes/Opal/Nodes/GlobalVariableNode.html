def compile
  helper :gvars

  if var_name == '&'
    return handle_global_match
  elsif var_name == "'"
    return handle_post_match
  elsif var_name == '`'
    return handle_pre_match
  end

  name = property var_name
  add_gvar name
  push "$gvars#{name}"
end

def handle_global_match
  with_temp do |tmp|
    push "((#{tmp} = $gvars['~']) === nil ? nil : #{tmp}['$[]'](0))"
  end
end
